71 void next()
72 {
73  char *pp;
74 
75  while (tk = *p) {
76  ++p;
77  if ((tk >= 'a' && tk <= 'z') || (tk >= 'A' && tk <= 'Z') || tk == '_') {
78  pp = p - 1;
79  while ((*p >= 'a' && *p <= 'z') || (*p >= 'A' && *p <= 'Z') || (*p >= '0' && *p <= '9') || *p == '_')
80  tk = tk * 147 + *p++;
81  tk = (tk << 6) + (p - pp);
82  id = sym;
83  while (id->tk) {
84  if (tk == id->hash && !memcmp(id->name, pp, p - pp)) { tk = id->tk; return; }
85  id = id + 1;
86  }
87  id->name = pp;
88  id->hash = tk;
89  tk = id->tk = Id;
90  return;
91  }
92  else if (tk >= '0' && tk <= '9') {
93  if (ival = tk - '0') { while (*p >= '0' && *p <= '9') ival = ival * 10 + *p++ - '0'; }
94  else if (*p == 'x' || *p == 'X') {
95  while ((tk = *++p) && ((tk >= '0' && tk <= '9') || (tk >= 'a' && tk <= 'f') || (tk >= 'A' && tk <= 'F')))
96  ival = ival * 16 + (tk & 15) + (tk >= 'A' ? 9 : 0);
97  }
98  else { while (*p >= '0' && *p <= '7') ival = ival * 8 + *p++ - '0'; }
99  tk = Num;
100  return;
101  }
102  switch (tk) {
103  case '\n':
104  if (src) {
105  printf("%d: %.*s", line, p - lp, lp);
106  lp = p;
107  while (le < e) {
108  printf("%8.4s", &ops[*++le * 5]);
109  if (*le <= ADJ) printf(" %d\n", *++le); else printf("\n");
110  }
111  }
112  ++line;
113  case ' ': case '\t': case '\v': case '\f': case '\r':
114  break;
115  case '/':
116  if (*p == '/') {
117  case '#':
118  while (*p != 0 && *p != '\n') ++p;
119  } else {
120  tk = Div;
121  return;
122  }
123  break;
124  case '\'':
125  case '"':
126  pp = data;
127  while (*p != 0 && *p != tk) {
128  if ((ival = *p++) == '\\') {
129  switch (ival = *p++) {
130  case 'n': ival = '\n'; break;
131  case 't': ival = '\t'; break;
132  case 'v': ival = '\v'; break;
133  case 'f': ival = '\f'; break;
134  case 'r': ival = '\r';
135  }
136  }
137  if (tk == '"') *data++ = ival;
138  }
139  ++p;
140  if (tk == '"') ival = (int)pp; else tk = Num;
141  return;
142  case '=': if (*p == '=') { ++p; tk = Eq; } else tk = Assign; return;
143  case '+': if (*p == '+') { ++p; tk = Inc; } else tk = Add; return;
144  case '-': if (*p == '-') { ++p; tk = Dec; } else if (*p == '>') { ++p; tk = Arrow; } else tk = Sub; return;
145  case '!': if (*p == '=') { ++p; tk = Ne; } return;
146  case '<': if (*p == '=') { ++p; tk = Le; } else if (*p == '<') { ++p; tk = Shl; } else tk = Lt; return;
147  case '>': if (*p == '=') { ++p; tk = Ge; } else if (*p == '>') { ++p; tk = Shr; } else tk = Gt; return;
148  case '|': if (*p == '|') { ++p; tk = Lor; } else tk = Or; return;
149  case '&': if (*p == '&') { ++p; tk = Lan; } else tk = And; return;
150  case '^': tk = Xor; return;
151  case '%': tk = Mod; return;
152  case '*': tk = Mul; return;
153  case '[': tk = Brak; return;
154  case '?': tk = Cond; return;
155  case '.': tk = Dot; return;
156  default: return;
157  }
158  }

161 void expr(int lev)
162 {
163  int t, *b, sz;
164  struct ident_s *d;
165  struct member_s *m;
166 
167  switch (tk) {
168  case 0: printf("%d: unexpected eof in expression\n", line); exit(-1);
169  case Num: *++e = IMM; *++e = ival; next(); ty = INT; break;
170  case '"':
171  *++e = IMM; *++e = ival; next();
172  while (tk == '"') next();
173  data = (char *)((int)data + sizeof(int) & -sizeof(int)); ty = PTR;
174  break;
175  case Sizeof:
176  next(); if (tk == '(') next(); else { printf("%d: open paren expected in sizeof\n", line); exit(-1); }
177  ty = INT; if (tk == Int) next(); else if (tk == Char) { next(); ty = CHAR; }
178  else if (tk == Struct) { next(); if (tk != Id) { printf("%d: bad struct type\n", line); exit(-1); } ty = id->stype; next(); }
179  while (tk == Mul) { next(); ty = ty + PTR; }
180  if (tk == ')') next(); else { printf("%d: close paren expected in sizeof\n", line); exit(-1); }
181  *++e = IMM; *++e = ty >= PTR ? sizeof(int) : tsize[ty];
182  ty = INT;
183  break;
184  case Id:
185  d = id; next();
186  if (tk == '(') {
187  next();
188  t = 0;
189  while (tk != ')') { expr(Assign); *++e = PSH; ++t; if (tk == ',') next(); }
190  next();
191  if (d->class == Sys) *++e = d->val;
192  else if (d->class == Fun) { *++e = JSR; *++e = d->val; }
193  else { printf("%d: bad function call\n", line); exit(-1); }
194  if (t) { *++e = ADJ; *++e = t; }
195  ty = d->type;
196  }
197  else if (d->class == Num) { *++e = IMM; *++e = d->val; ty = INT; }
198  else {
199  if (d->class == Loc) { *++e = LEA; *++e = loc - d->val; }
200  else if (d->class == Glo) { *++e = IMM; *++e = d->val; }
201  else { printf("%d: undefined variable\n", line); exit(-1); }
202  if ((ty = d->type) <= INT || ty >= PTR) *++e = (ty == CHAR) ? LC : LI;
203  }
204  break;
205  case '(':
206  next();
207  if (tk == Int || tk == Char || tk == Struct) {
208  if (tk == Int) { next(); t = INT; } else if (tk == Char) { next(); t = CHAR; }
209  else { next(); if (tk != Id) { printf("%d: bad struct type\n", line); exit(-1); } t = id->stype; next(); }
210  while (tk == Mul) { next(); t = t + PTR; }
211  if (tk == ')') next(); else { printf("%d: bad cast\n", line); exit(-1); }
212  expr(Inc);
213  ty = t;
214  }
215  else {
216  expr(Assign);
217  if (tk == ')') next(); else { printf("%d: close paren expected\n", line); exit(-1); }
218  }
219  break;
220  case Mul:
221  next(); expr(Inc);
222  if (ty > INT) ty = ty - PTR; else { printf("%d: bad dereference\n", line); exit(-1); }
223  if (ty <= INT || ty >= PTR) *++e = (ty == CHAR) ? LC : LI;
224  break;
225  case And:
226  next(); expr(Inc);
227  if (*e == LC || *e == LI) --e; // XXX else { printf("%d: bad address-of\n", line); exit(-1); }
228  ty = ty + PTR;
229  break;
230  case '!': next(); expr(Inc); *++e = PSH; *++e = IMM; *++e = 0; *++e = EQ; ty = INT; break;
231  case '~': next(); expr(Inc); *++e = PSH; *++e = IMM; *++e = -1; *++e = XOR; ty = INT; break;
232  case Add: next(); expr(Inc); ty = INT; break;
233  case Sub:
234  next(); *++e = IMM;
235  if (tk == Num) { *++e = -ival; next(); } else { *++e = -1; *++e = PSH; expr(Inc); *++e = MUL; }
236  ty = INT;
237  break;
238  case Inc: case Dec:
239  t = tk; next(); expr(Inc);
240  if (*e == LC) { *e = PSH; *++e = LC; }
241  else if (*e == LI) { *e = PSH; *++e = LI; }
242  else { printf("%d: bad lvalue in pre-increment\n", line); exit(-1); }
243  *++e = PSH;
244  *++e = IMM; *++e = ty >= PTR2 ? sizeof(int) : (ty >= PTR) ? tsize[ty - PTR] : 1;
245  *++e = (t == Inc) ? ADD : SUB;
246  *++e = (ty == CHAR) ? SC : SI;
247  break;
248  default: printf("%d: bad expression\n", line); exit(-1);
249  }
250 
251  while (tk >= lev) { // "precedence climbing" or "Top Down Operator Precedence" method
252  t = ty;
253  switch (tk) {
254  case Assign:
255  next();
256  if (*e == LC || *e == LI) *e = PSH; else { printf("%d: bad lvalue in assignment\n", line); exit(-1); }
257  expr(Assign); *++e = ((ty = t) == CHAR) ? SC : SI;
258  break;
259  case Cond:
260  next();
261  *++e = BZ; b = ++e;
262  expr(Assign);
263  if (tk == ':') next(); else { printf("%d: conditional missing colon\n", line); exit(-1); }
264  *b = (int)(e + 3); *++e = JMP; b = ++e;
265  expr(Cond);
266  *b = (int)(e + 1);
267  break;
268  case Lor: next(); *++e = BNZ; b = ++e; expr(Lan); *b = (int)(e + 1); ty = INT; break;
269  case Lan: next(); *++e = BZ; b = ++e; expr(Or); *b = (int)(e + 1); ty = INT; break;
270  case Or: next(); *++e = PSH; expr(Xor); *++e = OR; ty = INT; break;
271  case Xor: next(); *++e = PSH; expr(And); *++e = XOR; ty = INT; break;
272  case And: next(); *++e = PSH; expr(Eq); *++e = AND; ty = INT; break;
273  case Eq: next(); *++e = PSH; expr(Lt); *++e = EQ; ty = INT; break;
274  case Ne: next(); *++e = PSH; expr(Lt); *++e = NE; ty = INT; break;
275  case Lt: next(); *++e = PSH; expr(Shl); *++e = LT; ty = INT; break;
276  case Gt: next(); *++e = PSH; expr(Shl); *++e = GT; ty = INT; break;
277  case Le: next(); *++e = PSH; expr(Shl); *++e = LE; ty = INT; break;
278  case Ge: next(); *++e = PSH; expr(Shl); *++e = GE; ty = INT; break;
279  case Shl: next(); *++e = PSH; expr(Add); *++e = SHL; ty = INT; break;
280  case Shr: next(); *++e = PSH; expr(Add); *++e = SHR; ty = INT; break;
281  case Add:
282  next(); *++e = PSH; expr(Mul);
283  sz = (ty = t) >= PTR2 ? sizeof(int) : ty >= PTR ? tsize[ty - PTR] : 1;
284  if (sz > 1) { *++e = PSH; *++e = IMM; *++e = sz; *++e = MUL; }
285  *++e = ADD;
286  break;
287  case Sub:
288  next(); *++e = PSH; expr(Mul);
289  sz = t >= PTR2 ? sizeof(int) : t >= PTR ? tsize[t - PTR] : 1;
290  if (t == ty && sz > 1) { *++e = SUB; *++e = PSH; *++e = IMM; *++e = sz; *++e = DIV; ty = INT; }
291  else if (sz > 1) { *++e = PSH; *++e = IMM; *++e = sz; *++e = MUL; *++e = SUB; }
292  else *++e = SUB;
293  ty = t;
294  break;
295  case Mul: next(); *++e = PSH; expr(Inc); *++e = MUL; ty = INT; break;
296  case Div: next(); *++e = PSH; expr(Inc); *++e = DIV; ty = INT; break;
297  case Mod: next(); *++e = PSH; expr(Inc); *++e = MOD; ty = INT; break;
298  case Inc: case Dec:
299  if (*e == LC) { *e = PSH; *++e = LC; }
300  else if (*e == LI) { *e = PSH; *++e = LI; }
301  else { printf("%d: bad lvalue in post-increment\n", line); exit(-1); }
302  sz = ty >= PTR2 ? sizeof(int) : ty >= PTR ? tsize[ty - PTR] : 1;
303  *++e = PSH; *++e = IMM; *++e = sz;
304  *++e = (tk == Inc) ? ADD : SUB;
305  *++e = (ty == CHAR) ? SC : SI;
306  *++e = PSH; *++e = IMM; *++e = sz;
307  *++e = (tk == Inc) ? SUB : ADD;
308  next();
309  break;
310  case Dot:
311  ty = ty + PTR;
312  case Arrow:
313  if (ty <= PTR+INT || ty >= PTR2) { printf("%d: structure expected\n", line); exit(-1); }
314  next();
315  if (tk != Id) { printf("%d: structure member expected\n", line); exit(-1); }
316  m = members[ty - PTR]; while (m && m->id != id) m = m->next;
317  if (!m) { printf("%d: structure member not found\n", line); exit(-1); }
318  if (m->offset) { *++e = PSH; *++e = IMM; *++e = m->offset; *++e = ADD; }
319  ty = m->type;
320  if (ty <= INT || ty >= PTR) *++e = (ty == CHAR) ? LC : LI;
321  next();
322  break;
323  case Brak:
324  next(); *++e = PSH; expr(Assign);
325  if (tk == ']') next(); else { printf("%d: close bracket expected\n", line); exit(-1); }
326  if (t < PTR) { printf("%d: pointer type expected\n", line); exit(-1); }
327  sz = (t = t - PTR) >= PTR ? sizeof(int) : tsize[t];
328  if (sz > 1) { *++e = PSH; *++e = IMM; *++e = sz; *++e = MUL; }
329  *++e = ADD;
330  if ((ty = t) <= INT || ty >= PTR) *++e = (ty == CHAR) ? LC : LI;
331  break;
332  default: printf("%d: compiler error tk=%d\n", line, tk); exit(-1);
333  }
334  }

337 void stmt()
338 {
339  int *a, *b, *d, i;
340 
341  switch (tk) {
342  case If:
343  next();
344  if (tk == '(') next(); else { printf("%d: open paren expected\n", line); exit(-1); }
345  expr(Assign);
346  if (tk == ')') next(); else { printf("%d: close paren expected\n", line); exit(-1); }
347  *++e = BZ; b = ++e;
348  stmt();
349  if (tk == Else) {
350  *b = (int)(e + 3); *++e = JMP; b = ++e;
351  next();
352  stmt();
353  }
354  *b = (int)(e + 1);
355  return;
356  case While:
357  next();
358  a = e + 1;
359  if (tk == '(') next(); else { printf("%d: open paren expected\n", line); exit(-1); }
360  expr(Assign);
361  if (tk == ')') next(); else { printf("%d: close paren expected\n", line); exit(-1); }
362  *++e = BZ; b = ++e;
363  stmt();
364  *++e = JMP; *++e = (int)a;
365  *b = (int)(e + 1);
366  return;
367  case Switch:
368  next();
369  if (tk == '(') next(); else { printf("%d: open paren expected\n", line); exit(-1); }
370  expr(Assign);
371  if (tk == ')') next(); else { printf("%d: close paren expected\n", line); exit(-1); }
372  a = cas; *++e = JMP; cas = ++e;
373  b = brk; d = def; brk = def = 0;
374  stmt();
375  *cas = def ? (int)def : (int)(e + 1); cas = a;
376  while (brk) { a = (int *)*brk; *brk = (int)(e + 1); brk = a; }
377  brk = b; def = d;
378  return;
379  case Case:
380  *++e = JMP; ++e; *e = (int)(e + 7); *++e = PSH; i = *cas; *cas = (int)e;
381  next();
382  expr(Or);
383  if (e[-1] != IMM) { printf("%d: bad case immediate\n", line); exit(-1); }
384  *e = *e - i; *++e = SUB; *++e = BNZ; cas = ++e; *e = i + e[-3];
385  if (tk == ':') next(); else { printf("%d: colon expected\n", line); exit(-1); }
386  stmt();
387  return;
388  case Break:
389  next();
390  if (tk == ';') next(); else { printf("%d: semicolon expected\n", line); exit(-1); }
391  *++e = JMP; *++e = (int)brk; brk = e;
392  return;
393  case Default:
394  next();
395  if (tk == ':') next(); else { printf("%d: colon expected\n", line); exit(-1); }
396  def = e + 1;
397  stmt();
398  return;
399  case Return:
400  next();
401  if (tk != ';') expr(Assign);
402  *++e = LEV;
403  if (tk == ';') next(); else { printf("%d: semicolon expected\n", line); exit(-1); }
404  return;
405  case '{':
406  next();
407  while (tk != '}') stmt();
408  next();
409  return;
410  case ';':
411  next();
412  return;
413  default:
414  expr(Assign);
415  if (tk == ';') next(); else { printf("%d: semicolon expected\n", line); exit(-1); }
416  }
417 }

466  else if (tk == Char) { next(); bt = CHAR; }

470  if (tk == '{') {

496  }

501  while (tk != '}') {
502  mbt = INT;
503  if (tk == Int) next();
504  else if (tk == Char) { next(); mbt = CHAR; }
505  else if (tk == Struct) {
506  next();
507  if (tk != Id) { printf("%d: bad struct declaration\n", line); return -1; }
508  mbt = id->stype;
509  next();
510  }

512  ty = mbt;
